%------------------------------------------------------------------------------
% 17. Equation reference parentheses — Quarto 对公式交叉引用使用 \ref
%     而非 \eqref，导致编号不带括号。AtBeginDocument 在 hyperref 加载后执行，
%     确保 \ref 重定义生效（覆盖 hyperref 版本）。
%------------------------------------------------------------------------------
\usepackage{xstring}
\AtBeginDocument{%
  \let\quartoOldRef\ref
  \renewcommand{\ref}[1]{%
    \IfBeginWith{#1}{eq-}{%
      \textup{(\quartoOldRef{#1})}%
    }{%
      \quartoOldRef{#1}%
    }%
  }%
}
